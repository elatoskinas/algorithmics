	\begin{algorithm}
		\begin{algorithmic}
			\Procedure{DFS}{$G$, $v$}
				\State $ S \gets $ empty stack
				\State Push $ v $ to $ S $
				\State Mark $ v $ as visited
				\While{$ S \neq \emptyset $}
					\State $ u \gets $ pop top from $ S $
					\ForAll{$ c \in u.children $}
						\If{$ c $ is not visited}
							\State Mark $ c $ as visited	
							\State Push $ c $ to $ S $
						\EndIf
					\EndFor
				\EndWhile
			\EndProcedure
		\end{algorithmic}
	\end{algorithm}

	\begin{algorithm}
		\begin{algorithmic}
			\Procedure{DFS\_Recursive}{$G$, $v$}
				\State Mark $ v $ as visited
				\ForAll{$ c \in v.children $}
					\If{$ c $ is not visited}
						\State Recursively call $ DFS\_Recursive(G, c) $
					\EndIf
				\EndFor
			\EndProcedure
		\end{algorithmic}
	\end{algorithm}

	\begin{algorithm}
		\begin{algorithmic}
			\Procedure{BFS}{$G$, $v$}
				\State $ Q \gets $ empty queue
				\State Enqueue $ v $ to $ Q $
				\State Mark $ v $ as visited
				\While{$ Q \neq \emptyset $}
					\State $ u \gets $ dequeue from $ Q $
					\ForAll{$ c \in u.children $}
						\If{$ c $ is not visited}
						\State Mark $ c $ as visited	
						\State Enqueue $ c $ to $ Q $
						\EndIf
					\EndFor
				\EndWhile
			\EndProcedure
		\end{algorithmic}
	\end{algorithm}
