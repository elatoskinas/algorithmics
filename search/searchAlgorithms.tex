	\begin{algorithm}
		\textbf{Input:} Sorted array, size of array and element to search for
		\\
		\textbf{Output:} Index of element in array if found, or -1
		\begin{algorithmic}
			\Procedure{Binary\_Search}{$A$, $n$, $el$}
				\State $ L \gets 0 $
				\State $ R \gets n-1 $
				\While{$ L \leq R $}
					\State $ m \gets \lfloor (L + R)/2 \rfloor $
					\If{$ A[m] = el $}
						\Return $ m $
					\ElsIf{$ A[m] < el $}
						\State $ l \gets m + 1 $
					\Else
						\State $ r \gets m - 1 $
					\EndIf
				\EndWhile
				\Return $ -1 $
			\EndProcedure
		\end{algorithmic}
	\end{algorithm}

		\\\Comment Note: Used for unimodal functions.
		\begin{algorithmic}
			\Procedure{Ternary\_Search}{$f$, $left$, $right$, $precision$}
				\If{$ |right - left| < precision $}
					\State \Return $ (left + right) / 2 $
				\EndIf

				\State $ left\_third \gets (2*left + right) / 3 $
				\State $ right\_third \gets (left + 2*right) / 3 $
